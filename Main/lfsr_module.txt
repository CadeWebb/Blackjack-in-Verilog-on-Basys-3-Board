module LFSR(input clk, output[31:0] q);
	reg [31:0] rQ = 0;
	reg start;
	reg first = 0;
	always @ (posedge clk)
	    begin
		if (first == 0) begin
		 rQ <= 32'b01010101101010101111111100000000; //HARD CODED
		 first <= 1;
		end
		else
		 rQ <= {rQ[31:0], start};
		end

	always @ (*)
	    begin
	    start = rQ[0] ^ rQ[10] ^ rQ[30] ^ rQ[31];
	    end

	assign q = rQ[31:0];
endmodule
